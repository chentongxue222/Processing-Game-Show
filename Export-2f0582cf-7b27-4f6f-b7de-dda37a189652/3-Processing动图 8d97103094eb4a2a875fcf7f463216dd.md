# 3-Processing动图

### 参考网站：

[老师，我再也不怕Processing动图啦 - 深度解析Processing图片序列帧动画](https://mp.weixin.qq.com/s/O1awDaha3M4SaEEBqJq0Yw)

```java
import gifAnimation.*;

PImage[] animation;
Gif loopingGif;

boolean pause = false;

void setup()
{
 size(800,300);
 frameRate(60);
 println("gifAnimation " + Gif.version());
 loopingGif = new Gif(this, "demo1.gif");
 loopingGif.loop();
 animation = Gif.getPImages(this, "demo1.gif");
}

void draw()
{
  background(255);
  imageMode(CENTER);
  image(loopingGif,width/2,height/2,100,100);
}

void keyPressed()    //控制动图的播放和暂停
{
 if(key == ' ')
 {
   pause = !pause;
  if(pause == false)
  {
    loopingGif.pause();
  }
  else if(pause == true)
  {
    loopingGif.play();
  }
 }
}
```
